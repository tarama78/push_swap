speed:
	before:
		10000 -> 238000
		500 -> 7600
		100 -> 1070
		10 -> 40
		5 -> 15
	after:
		10000 -> 183000
		500 -> 6000
		100 -> 820
		10 -> 38
		5 -> 15

sort 2 or 3 int:
	before:
		2 int -> 3
		3 int -> 8
	after:
		2 int -> 3 (-0)
		3 int -> 6 (-2)

sort block:
	before:
		size / 2 * 2 + size / 2 + size % 2 + size (~size * 2.5)
	after:
		(size / 2 + size % 2) * 2 + size / 2 (~size * 1.5)////////////
